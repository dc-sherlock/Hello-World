#include<iostream>
#include<math.h>
using namespace std;

int main()
{
   char ss[10]={'0'};
   int i;
   
   for(i=0;i<10;i++) ss[i]='0';
   cin.getline(ss,10);
   for(i=0;ss[i]!='0';i++)
   {
      if((i%2==1)&&ss[i]>'a'&&ss[i]<'z') ss[i]+=('A'-'a');          
   }
   
   for(i=0;ss[i]!='0';i++)
   cout<<ss[i];
   
   system("pause");    
}
